const http = require('http'); // Create an HTTP server that listens on port 3000 const server = http.createServer((req, res) => { // Set the HTTP response header with a 200 status code and plain text content type res.writeHead(200, { 'Content-Type': 'text/plain' }); // Write the "Hello, World!" message to the response res.end('Hello, World!\n'); }); // Listen on port 3000 const PORT = 3000; server.listen(PORT, () => { console.log(`Server is running on http://localhost:${PORT}`); });
